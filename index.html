/** * ============================================ * SKILLQUEST — ГЛАВНЫЙ ФАЙЛ ПРИЛОЖЕНИЯ * ============================================ * Назначение: управление календарём, пользователем, * темами, меню и базовой логикой. * ============================================ */ // ============================================ // 1. ДАННЫЕ ПОЛЬЗОВАТЕЛЯ (хранятся в localStorage) // ============================================ /** Объект с данными пользователя */ const userData = { name: 'Игрок', level: 1, exp: 0, avatar: 'assets/default-avatar.png', // путь к аватарке }; /** Ключ для сохранения в localStorage */ const STORAGE_KEY = 'skillquest_user'; /** * Загружает данные пользователя из localStorage * Если данных нет — создаёт дефолтные */ function loadUserData() { const saved = localStorage.getItem(STORAGE_KEY); if (saved) { try { const parsed = JSON.parse(saved); Object.assign(userData, parsed); } catch (e) { console.warn('Ошибка загрузки данных, используем дефолтные'); } } updateUserUI(); } /** * Сохраняет данные пользователя в localStorage */ function saveUserData() { localStorage.setItem(STORAGE_KEY, JSON.stringify(userData)); } /** * Обновляет интерфейс карточки пользователя */ function updateUserUI() { document.getElementById('userName').textContent = userData.name; document.getElementById('userLevel').textContent = userData.level; document.getElementById('userExp').textContent = userData.exp; document.getElementById('userAvatar').src = userData.avatar; } // ============================================ // 2. КАЛЕНДАРЬ // ============================================ let currentDate = new Date(); // текущая дата (для навигации) let currentMode = 'month'; // 'day' | 'week' | 'month' | 'year' /** Элементы календаря */ const calendarGrid = document.getElementById('calendarGrid'); const monthTitle = document.getElementById('monthTitle'); /** * Отрисовывает календарь на месяц * @param {number} year - год * @param {number} month - месяц (0-11) */ function renderMonthCalendar(year, month) { // Очищаем сетку, но оставляем заголовки дней недели const weekdays = calendarGrid.querySelectorAll('.calendar-grid__weekday'); calendarGrid.innerHTML = ''; weekdays.forEach(el => calendarGrid.appendChild(el)); // Первый день месяца const firstDay = new Date(year, month, 1); // Последний день месяца const lastDay = new Date(year, month + 1, 0); const daysInMonth = lastDay.getDate(); // День недели первого числа (0 - воскресенье, 1 - понедельник) // Сдвигаем, чтобы неделя начиналась с понедельника let startOffset = firstDay.getDay() - 1; if (startOffset < 0) startOffset = 6; // если воскресенье // Заполняем пустые ячейки до первого дня for (let i = 0; i < startOffset; i++) { const empty = document.createElement('div'); empty.className = 'calendar-day calendar-day--other-month'; calendarGrid.appendChild(empty); } // Дни месяца const today = new Date(); const todayDate = today.getDate(); const todayMonth = today.getMonth(); const todayYear = today.getFullYear(); for (let d = 1; d <= daysInMonth; d++) { const dayEl = document.createElement('div'); dayEl.className = 'calendar-day'; dayEl.textContent = d; // Подсветка сегодняшнего дня if (d === todayDate && month === todayMonth && year === todayYear) { dayEl.classList.add('calendar-day--today'); } // TODO: добавить маркеры задач (кружочки) // if (hasTask(year, month, d)) { dayEl.dataset.hasTask = 'true'; } calendarGrid.appendChild(dayEl); } // Заголовок месяца const monthNames = ['Январь', 'Февраль', 'Март', 'Апрель', 'Май', 'Июнь', 'Июль', 'Август', 'Сентябрь', 'Октябрь', 'Ноябрь', 'Декабрь']; monthTitle.textContent = `${monthNames[month]} ${year}`; } /** * Переключает месяц (вперёд/назад) */ function changeMonth(delta) { currentDate.setMonth(currentDate.getMonth() + delta); renderMonthCalendar(currentDate.getFullYear(), currentDate.getMonth()); } // ============================================ // 3. ПЕРЕКЛЮЧЕНИЕ ТЕМЫ (День/Ночь) // ============================================ const themeToggle = document.getElementById('themeToggle'); const htmlElement = document.documentElement; /** Текущая тема: 'light' или 'dark' */ let currentTheme = localStorage.getItem('skillquest_theme') || 'light'; function applyTheme(theme) { htmlElement.setAttribute('data-theme', theme); themeToggle.textContent = theme === 'light' ? '🌙' : '☀️'; localStorage.setItem('skillquest_theme', theme); currentTheme = theme; } themeToggle.addEventListener('click', () => { const newTheme = currentTheme === 'light' ? 'dark' : 'light'; applyTheme(newTheme); }); // ============================================ // 4. МЕНЮ (САЙДБАР) // ============================================ const menuToggle = document.getElementById('menuToggle'); const sidebar = document.getElementById('sidebar'); const overlay = document.getElementById('sidebarOverlay'); function openSidebar() { sidebar.classList.add('sidebar--open'); } function closeSidebar() { sidebar.classList.remove('sidebar--open'); } menuToggle.addEventListener('click', openSidebar); overlay.addEventListener('click', closeSidebar); // Закрытие по Escape document.addEventListener('keydown', (e) => { if (e.key === 'Escape') closeSidebar(); }); // ============================================ // 5. ОБРАБОТЧИКИ ПУНКТОВ МЕНЮ (заглушки) // ============================================ document.querySelectorAll('.sidebar__item').forEach(item => { item.addEventListener('click', () => { const page = item.dataset.page; console.log(`📱 Открыт пункт: ${page}`); closeSidebar(); // TODO: открыть соответствующее модальное окно alert(`Пункт "${page}" — будет реализован позже`); }); }); // ============================================ // 6. ПЕРЕКЛЮЧЕНИЕ РЕЖИМОВ КАЛЕНДАРЯ // ============================================ document.querySelectorAll('.calendar-mode__btn').forEach(btn => { btn.addEventListener('click', () => { // Снимаем активный класс со всех document.querySelectorAll('.calendar-mode__btn').forEach(b => { b.classList.remove('calendar-mode__btn--active'); }); btn.classList.add('calendar-mode__btn--active'); const mode = btn.dataset.mode; currentMode = mode; console.log(`📅 Режим календаря: ${mode}`); // TODO: реализовать переключение между Day/Week/Month/Year if (mode === 'month') { renderMonthCalendar(currentDate.getFullYear(), currentDate.getMonth()); } else { // Заглушка для других режимов alert(`Режим "${mode}" в разработке`); } }); }); // ============================================ // 7. ИНИЦИАЛИЗАЦИЯ ПРИЛОЖЕНИЯ // ============================================ function initApp() { // Загружаем данные пользователя loadUserData(); // Применяем сохранённую тему applyTheme(currentTheme); // Рендерим календарь на текущий месяц renderMonthCalendar(currentDate.getFullYear(), currentDate.getMonth()); // Навигация по месяцам document.getElementById('prevMonth').addEventListener('click', () => changeMonth(-1)); document.getElementById('nextMonth').addEventListener('click', () => changeMonth(1)); console.log('🚀 SkillQuest запущен!'); console.log(`👤 Пользователь: ${userData.name}, Уровень: ${userData.level}`); } // Запускаем приложение после загрузки DOM document.addEventListener('DOMContentLoaded', initApp);
